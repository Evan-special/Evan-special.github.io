# 特定值出现的次数

const countOccurrences = (array, value) => {
  return array.reduce((arr, val) => value === val ? arr + 1 : arr + 0, 0);
}

# 乱序

const shuffle = arr => arr.sort(() => Math.random() - 0.5); // 非概率平均
const shuffle = arr => {
  let len = arr.length;
  while(len) {
    let index = Math.floor(Math.random() * len--);
    let current = arr[len];
    arr[len] = arr[index];
    arr[index] = current;
  }
  return arr;
}
